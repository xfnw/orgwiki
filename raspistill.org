#+begin_src sh :tangle full.cgi
#!/bin/sh
echo "Content-Type: image/jpeg"
echo
raspistill -o - -t 1 -n
#+end_src

echo
raspistill -o - -t 1 -n -w 320 -h 240
#+end_src
